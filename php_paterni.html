Factory patern

Jedan od najčešće korišćenih paterna

Objekti se kreiraju bez otkrivanja logike kreiranja klijentu

Factory klasa

            
class Mailbox_Display_Factory
{
    /**
    * Returns new instance of inbox display printer
    *
    * @param string $type
    * @param array $config
    */
    public static function forgeDisplayPrinter($type, array $config = array())
    {
        $class_name = '\\Application\\Message\\Mailbox_Display_'.ucfirst($type);

        return new $class_name($config);
    }
}

Strategy klase

            
class Mailbox_Display_Abstract
{
    public function __construct(array $config = array())
    {
        $this->config = $config;
    }

    public function render()
    {
        $this->prepare_data();
        $this->view = View::forge('messages/templates/messages/wrapper/'.$this->view_name, $this->view_data, false);

        return $this->view->render();
    }

    protected function prepareData()
    {
        return false;
    }
}
